function [r] = PoliNewton(m,x,y,z) %(número de pontos, abcissas, ordenadas, valor a interpolar)
  r=0;
  for i=1:m
  dely(i)=y(i);
 endfor 
    for k=1:m-1
      for i=m:-1:k+1
         dely(i)=(dely(i)-dely(i-1))/(x(i)-x(i-k)); %definido na tabela das diferenças divididas
      endfor
   endfor   
   r=dely(m);
     for i=m-1:-1:1
       r=r*(z-x(i))+dely(i);
     endfor  
   
  display('O valor de z interpolado é r')
endfunction
